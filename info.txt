<VirtualHost *:443>
    ServerName portfolio.nigerdev.com

    DocumentRoot /var/www/port/out

    <Directory /var/www/port/out>
        Options -Indexes +FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>

    # Activer mod_headers pour le cache
    <IfModule mod_headers.c>
        Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains"
        Header always set X-Content-Type-Options "nosniff"
        Header always set X-XSS-Protection "1; mode=block"
        Header always set X-Frame-Options "SAMEORIGIN"
    </IfModule>

    SSLEngine On
    SSLCertificateFile /etc/letsencrypt/live/portfolio.nigerdev.com/fullchain.pem
    SSLCertificateKeyFile /etc/letsencrypt/live/portfolio.nigerdev.com/privkey.pem
    Include /etc/letsencrypt/options-ssl-apache.conf

    ErrorLog ${APACHE_LOG_DIR}/portfolio.nigerdev.com_ssl_error.log
    CustomLog ${APACHE_LOG_DIR}/portfolio.nigerdev.com_ssl_access.log combined
</VirtualHost>
